CREATE SCHEMA IF NOT EXISTS `giphy` DEFAULT CHARACTER SET utf8 ;
USE `giphy` ;
CREATE TABLE IF NOT EXISTS `giphy`.`image` (
  `transId` INT(20) NOT NULL AUTO_INCREMENT,
  `imageId` VARCHAR(50) NULL DEFAULT NULL,
  `email` VARCHAR(255) NOT NULL,
  PRIMARY KEY (`transId`))
ENGINE = InnoDB
AUTO_INCREMENT = 1
DEFAULT CHARACTER SET = utf8;

CREATE UNIQUE INDEX `transId_UNIQUE` ON `giphy`.`image` (`transId` ASC);
